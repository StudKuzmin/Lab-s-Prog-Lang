{
	int count = 0;

	for (int i = 0; i < f; i++)
	{
		if (mas_bus[i].FIO == "Petrov_Ivan_Alekseevich")
		{
			cout << "Маршрут(ы) Петрова: " << mas_bus[i].route_number << endl;
			count++;
		}
		if (i == f - 1 && count == 0) cout << "Нет водителя с фамилией Петров." << endl;
	}

}
