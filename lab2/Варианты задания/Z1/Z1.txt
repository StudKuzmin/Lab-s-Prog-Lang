#include <ctime>
#include <iostream>
#include <windows.h>
#include <string>
#include "stdio.h"


using namespace std;

int main()
{
    setlocale(LC_ALL, "ru");
    
    int** m;
    int rows, cols;

    cout << "Введите количество строк(рекомендуется использовать только нечётные числа): ";
    cin >> rows;
    cout << "Введите количество столбцов(рекомендуется использовать только нечётные числа): ";
    cin >> cols;

    m = new int* [rows];

    int a, b;
    cout << "Введите значения интервала [a, b]: " << endl;
    cin >> a >> b;

    if (a > b)//Меняем местами элементы
    {
        a = a + b;
        b = a - b;
        a = a - b;
    }
    printf("Ваш диапазон: [%d, %d]", a, b);
    cout << endl;

    for (int i = 0; i < rows; i++)
    {
        m[i] = new int[cols];//Присваиваем элементам указателей новые динамические массивы размером количества СТОЛБЦОВ		
    }

    for (int i = 0; i < rows; i++)//Заполняем двумерный динамический массив числами
    {
        for (int j = 0; j < cols; j++)
        {
            m[i][j] = rand() % (b - a + 1) + a;//[a, b]
        }
    }

    for (int i = 0; i < rows; i++)//Выводим массив на экран
    {
        for (int j = 0; j < cols; j++)
        {
            cout << m[i][j] << "\t";
            
        }
        cout << endl << endl << endl;
    }

    

}
