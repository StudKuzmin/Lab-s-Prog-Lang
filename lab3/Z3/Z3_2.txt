void Razdelenie(char* str, char* str1, int lenghtStr)//Разделение строк на слова и обновление массива

        {
            str[i] = '\n';
        }
    }

    int j = 0;
    for (int i = 0; i < lenghtStr; i++, j++)

        if (str[i] == '\n' && str[i + 1] == '\n')
        {
            i++;
        }
    }
    str1[j] = '\0';
}
