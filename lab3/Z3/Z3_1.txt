void Count_Symbols_Words(char* str, char* word, int lenghtStr)//Ищем самое длинное слово и выводим его на экран
{
    int max = 0;
    int count = 0;
    int ws = 0;
    for (int i = 0; i < lenghtStr; i++)
    {
        if (str[i] == '\n') count = 0;
        if (str[i] != '\n')
        {
            count++;
            if (max < count)
            {
                max = count;
                ws = i;
            }
        }

    }

    int j = 0;
    ws++;
    for (int i = ws - max; i <= ws; i++)
    {
        word[j] = str[i];
        j++;
    }
    word[j] = '\0';

    cout << "Самое длинное слово в тексте: " << word;
    cout << "Длина слова: " << max << endl << endl;
}

void Razdelenie(char* str, char* str1, int lenghtStr)//Разделение строк на слова
{
    for (int i = 0; i < lenghtStr; i++)
    {
        if (str[i] == ' ' || str[i] == '?' || str[i] == '!' || str[i] == '.')
        {
            str[i] = '\n';
        }
    }

    int j = 0;
    for (int i = 0; i < lenghtStr; i++, j++)
    {
        str1[j] = str[i];
        if (str[i] == '\n' && str[i + 1] == '\n')
        {
            i++;
        }
    }
    str1[j] = '\0';
}
