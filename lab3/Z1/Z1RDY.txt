#include <ctime>
#include <iostream>
#include <windows.h>
#include "stdio.h"

using namespace std;


void StrSum(char* str3, char str1[255], char str2[255], int n, int m, int size)
{
    for (int i = 0; i < n; i++)
    {
        str3[i] = str1[i];
    }
    
    str3[n] = ' ';
    n++;

    for (int i = 0; i < m; i++)
    {
        str3[n] = str2[i];
        n++;
    }

    str3[size+1] = '\0';
}

void main()
{
    setlocale(LC_ALL, "ru");
    srand(time(0));

    char str1[255];
    char str2[255];

    cout << "Введите строку 1: "; gets_s(str1);
    cout << "Введите строку 2: "; gets_s(str2);

    int n = 0;
    while (true)
    {
        if (str1[n] == '\0') break;
        n++;
    }
    cout << endl << "Количество символов str1: " << n << endl;

    int m = 0;
    while (true)
    {
        if (str2[m] == '\0') break;
        m++;
    }
    cout << "Количество символов str2: " << m << endl;

    int size = n + m;
    char* str3 = new char[size+1];

    
    StrSum(str3, str1, str2, n, m, size);

    cout << endl << str3;

}
