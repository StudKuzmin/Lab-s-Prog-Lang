#include <iostream>
#include <ctime>

using namespace std;

int RekFact(int n) // Рекурсивная функция
{
	if (n == 0 || n == 1) return 1; // Частный случай, выход из рекурсии
	if (n == 2) return 2;           // Частный случай, выход из рекурсии

	return n * RekFact(n - 1);
}

int Fact(int n) // Не рекурсивная функция
{
	int fac = 1;
	for (int i = 1; i <= n; i++)
	{
		fac = fac * i;
	}

	return fac;
}

int main()
{
	setlocale(LC_ALL, "ru"); srand(time(NULL));

	int n, f;

	cout << "Факториал скольких чисел вы хотите найти? "; cin >> f;
	cout << endl;

	for (n = 0; n < f; n++)
	{
		printf("рекурсивная: %d! = %d \t не рекурсивная: %d! = %d\n", n, RekFact(n), n, Fact(n));
	}

	return 0;
}
