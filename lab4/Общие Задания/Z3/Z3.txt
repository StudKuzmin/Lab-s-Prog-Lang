#include <iostream>
#include <ctime>

using namespace std;

int step(int i) // Функция степеней числа 2
{
	if (i == 0) return 1; // Частный случай
	if (i == 1) return 2; // Частный случай

	return 2 * step(i - 1);
}

int main()
{
	setlocale(LC_ALL, "ru"); srand(time(NULL));

	int N;

	cout << "Введите число: "; cin >> N;
	cout << endl;

	for (int i = 0; true; i++)
	{
		if (N == step(i))
		{
			printf("YES: %d = 2^%d\n", N, i);
			break;
		}

		else if (i == 30) // Ограничиваем нашу проверку до 30 степени
		{
			cout << "NO" << endl;
			break;
		}
	}


	return 0;
}
