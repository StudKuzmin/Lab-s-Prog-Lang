#include <iostream>
#include <ctime>

using namespace std;

int rek(int n)
{
	if (n == 1) return 3; // Частный случай, выход из рекурсии
	
	return (n - 1) * rek(n - 1);
}

int main()
{
	setlocale(LC_ALL, "ru"); srand(time(NULL));

	int n, f;

	cout << "Ограничение: "; cin >> f;
	cout << endl;

	for (n = 1; n <= f; n++) printf("F(%d) = %d\n", n, rek(n));

	return 0;
}
